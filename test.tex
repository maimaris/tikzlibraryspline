\newcommand*\showcontrols{[%
  spline coordinates,
  execute at end to={%
    {%
    \foreach \i in {1,2,...,\tikzsplinesegments}{%
      \draw[black,-] let \n0={\i+1} in
        (spline K-\i) -- (spline P-\i)
        (spline Q-\i) -- (spline K-\n0);
      \fill[green!50!black] (spline P-\i) circle (1pt);
      \fill[red] (spline Q-\i) circle (1pt);
    };%
    }%
  }%
]}
